	private void HandleDecision()
	{
		if (!this.team.IsPlayersTeam())
		{
			this.dateTeamReplied = Game.instance.time.now;
			if (Game.instance.time.now.Month == 12)
			{
				this.mStatus = PlayerJobApplication.Status.WaitingPlayerReply;
			}
			else if (this.GetPlayerLoyaltyDealValue() + this.GetPlayerFinancesDealValue() + this.GetTeamChairmanHappinessDealValue() + this.GetTeamMarketabilityDealValue() + this.GetGPDifferenceDealValue() > this.mDecisionThreshold || Game.instance.player.hasJoinedTeamRecently)
			{
				this.mStatus = PlayerJobApplication.Status.Declined;
			}
			else
			{
				this.mStatus = PlayerJobApplication.Status.WaitingPlayerReply;
			}
			if (this.mPreSeason)
			{
				this.message = Game.instance.dialogSystem.OnPlayerJobApplicationPreSeasonDecision(this.team.chairman, this.isSuccessfull);
				return;
			}
			this.message = Game.instance.dialogSystem.OnPlayerJobApplicationDecision(this.team.chairman, this.isSuccessfull);
		}
	}
