	private void SetAvailableBackstories()
	{
		List<PlayerBackStory.PlayerBackStoryType> list = new List<PlayerBackStory.PlayerBackStoryType>();
		int num = 7;
		for (int i = 0; i < num; i++)
		{
			PlayerBackStory.PlayerBackStoryType playerBackStoryType = (PlayerBackStory.PlayerBackStoryType)i;
			if (PlayerBackStory.isBackstoryUnlocked(playerBackStoryType))
			{
				list.Add(playerBackStoryType);
			}
		}
		this.mAvailableBackstories = list.ToArray();
	}
